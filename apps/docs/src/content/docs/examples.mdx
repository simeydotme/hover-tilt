## Basic Usage

The simplest usage with default props. The component will apply a subtle tilt effect on hover.

<BasicDemo client:only="svelte" />

<Tabs>
	<TabItem label="Svelte">
		```svelte
		<script>
		  import { HoverTilt } from 'hover-tilt';
		</script>

    	<HoverTilt>
    	  <div class="card">
    	    <h3>Basic Example</h3>
    	    <p>Hover over this card to see the default tilt effect</p>
    	  </div>
    	</HoverTilt>
    	```

## Tilt Factors

Control the intensity of the tilt effect with `tiltFactor` and `tiltFactorY`. The `tiltFactor` controls horizontal tilt, while `tiltFactorY` controls vertical tilt (defaults to `tiltFactor` if not specified).

<TiltDemo client:only="svelte" />

<Tabs>
	<TabItem label="Svelte">
		```svelte
		<script>
		  import { HoverTilt } from 'hover-tilt';
		</script>

    	<!-- Default tilt -->
    	<HoverTilt>
    	  <div class="card">tiltFactor: 1</div>
    	</HoverTilt>

    	<!-- Increased tilt -->
    	<HoverTilt tiltFactor={2}>
    	  <div class="card">tiltFactor: 2</div>
    	</HoverTilt>

    	<!-- Different Y tilt -->
    	<HoverTilt tiltFactor={1.5} tiltFactorY={0.5}>
    	  <div class="card">tiltFactor: 1.5, tiltFactorY: 0.5</div>
    	</HoverTilt>
    	```
    </TabItem>
    <TabItem label="Web Component">
    	```html
    	<!-- Default tilt -->
    	<hover-tilt>
    	  <div class="card">tilt-factor: 1</div>
    	</hover-tilt>

    	<!-- Increased tilt -->
    	<hover-tilt tilt-factor="2">
    	  <div class="card">tilt-factor: 2</div>
    	</hover-tilt>

    	<!-- Different Y tilt -->
    	<hover-tilt tilt-factor="1.5" tilt-factor-y="0.5">
    	  <div class="card">tilt-factor: 1.5, tilt-factor-y: 0.5</div>
    	</hover-tilt>
    	```
    </TabItem>

</Tabs>

## Spring Animations

Customize the animation physics with `springOptions` and `tiltSpringOptions`. Control stiffness, damping, and precision for different animation feels.

<SpringDemo client:only="svelte" />

<Tabs>
	<TabItem label="Svelte">
		```svelte
		<script>
		  import { HoverTilt } from 'hover-tilt';
		</script>

    	<!-- Default spring -->
    	<HoverTilt>
    	  <div class="card">Default</div>
    	</HoverTilt>

    	<!-- Fast spring -->
    	<HoverTilt springOptions={{ stiffness: 0.5, damping: 0.6 }}>
    	  <div class="card">stiffness: 0.5, damping: 0.6</div>
    	</HoverTilt>

    	<!-- Slow spring -->
    	<HoverTilt springOptions={{ stiffness: 0.1, damping: 0.9 }}>
    	  <div class="card">stiffness: 0.1, damping: 0.9</div>
    	</HoverTilt>

    	<!-- Custom tilt spring -->
    	<HoverTilt tiltSpringOptions={{ stiffness: 0.3, damping: 0.7 }}>
    	  <div class="card">tiltSpringOptions</div>
    	</HoverTilt>
    	```
    </TabItem>
    <TabItem label="Web Component">
    	```html
    	<!-- Default spring -->
    	<hover-tilt>
    	  <div class="card">Default</div>
    	</hover-tilt>

    	<!-- Fast spring (requires JSON string) -->
    	<hover-tilt spring-options='{"stiffness": 0.5, "damping": 0.6}'>
    	  <div class="card">stiffness: 0.5, damping: 0.6</div>
    	</hover-tilt>

    	<!-- Slow spring -->
    	<hover-tilt spring-options='{"stiffness": 0.1, "damping": 0.9}'>
    	  <div class="card">stiffness: 0.1, damping: 0.9</div>
    	</hover-tilt>

    	<!-- Custom tilt spring -->
    	<hover-tilt tilt-spring-options='{"stiffness": 0.3, "damping": 0.7}'>
    	  <div class="card">tilt-spring-options</div>
    	</hover-tilt>
    	```
    </TabItem>

</Tabs>

## Exit Delay

Control how long the component waits before animating back to the default state when the cursor leaves. Use `exitDelay` (in milliseconds) to create a smoother experience.

<Tabs>
	<TabItem label="Svelte">
		```svelte
		<script>
		  import { HoverTilt } from 'hover-tilt';
		</script>

    	<!-- Default exit delay (200ms) -->
    	<HoverTilt>
    	  <div class="card">Default</div>
    	</HoverTilt>

    	<!-- Longer exit delay -->
    	<HoverTilt exitDelay={500}>
    	  <div class="card">exitDelay: 500ms</div>
    	</HoverTilt>

    	<!-- No exit delay -->
    	<HoverTilt exitDelay={0}>
    	  <div class="card">exitDelay: 0ms</div>
    	</HoverTilt>
    	```
    </TabItem>
    <TabItem label="Web Component">
    	```html
    	<!-- Default exit delay (200ms) -->
    	<hover-tilt>
    	  <div class="card">Default</div>
    	</hover-tilt>

    	<!-- Longer exit delay -->
    	<hover-tilt exit-delay="500">
    	  <div class="card">exit-delay: 500ms</div>
    	</hover-tilt>

    	<!-- No exit delay -->
    	<hover-tilt exit-delay="0">
    	  <div class="card">exit-delay: 0ms</div>
    	</hover-tilt>
    	```
    </TabItem>

</Tabs>

## Blend Modes

Change how the glare effect blends with the content using the `blendMode` prop. Common values include `overlay`, `screen`, `multiply`, `soft-light`, and more.

<Tabs>
	<TabItem label="Svelte">
		```svelte
		<script>
		  import { HoverTilt } from 'hover-tilt';
		</script>

    	<!-- Default blend mode (overlay) -->
    	<HoverTilt>
    	  <div class="card">Default (overlay)</div>
    	</HoverTilt>

    	<!-- Screen blend mode -->
    	<HoverTilt blendMode="screen">
    	  <div class="card">blendMode: screen</div>
    	</HoverTilt>

    	<!-- Soft light blend mode -->
    	<HoverTilt blendMode="soft-light">
    	  <div class="card">blendMode: soft-light</div>
    	</HoverTilt>
    	```
    </TabItem>
    <TabItem label="Web Component">
    	```html
    	<!-- Default blend mode (overlay) -->
    	<hover-tilt>
    	  <div class="card">Default (overlay)</div>
    	</hover-tilt>

    	<!-- Screen blend mode -->
    	<hover-tilt blend-mode="screen">
    	  <div class="card">blend-mode: screen</div>
    	</hover-tilt>

    	<!-- Soft light blend mode -->
    	```
