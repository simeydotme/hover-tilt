## Basic Usage

The simplest usage with default props. The component will apply a subtle tilt effect on hover.

## Tilt Factors

Control the intensity of the tilt effect with `tiltFactor` and `tiltFactorY`.
The `tiltFactor` controls tilt on both axes, while `tiltFactorY` overrides control of the vertical tilt.

<TiltDemo client:only="svelte" />

<Tabs>
	<TabItem label="Svelte">
		```svelte

    	<!-- Default tilt -->
    	<HoverTilt>
    	  <div class="card">tiltFactor: 1</div>
    	</HoverTilt>

    	<!-- Increased tilt -->
    	<HoverTilt tiltFactor={2}>
    	  <div class="card">tiltFactor: 2</div>
    	</HoverTilt>

    	<!-- Different Y tilt -->
    	<HoverTilt tiltFactor={1.5} tiltFactorY={0.5}>
    	  <div class="card">tiltFactor: 1.5, tiltFactorY: 0.5</div>
    	</HoverTilt>
    	```
    </TabItem>
    <TabItem label="Web Component">
    	```html
    	<!-- Default tilt -->
    	<hover-tilt>
    	  <div class="card">tilt-factor: 1</div>
    	</hover-tilt>

    	<!-- Increased tilt -->
    	<hover-tilt tilt-factor="2">
    	  <div class="card">tilt-factor: 2</div>
    	</hover-tilt>

    	<!-- Different Y tilt -->
    	<hover-tilt tilt-factor="1.5" tilt-factor-y="0.5">
    	  <div class="card">tilt-factor: 1.5, tilt-factor-y: 0.5</div>
    	</hover-tilt>
    	```
    </TabItem>

</Tabs>

## Spring Animations

Customize the animation physics with `springOptions` and `tiltSpringOptions`.
Control stiffness & damping for different feeling interactions.

- A stiffer spring will create more pronouced elastic effects (more 'boing-y')
- More damping will smooth out the motion at the start and end (less 'zippy')

<SpringDemo client:only="svelte" />

<Tabs>
	<TabItem label="Svelte">
		```svelte

    	<!-- Default spring -->
    	<HoverTilt>
    	  <div class="card">Default</div>
    	</HoverTilt>

    	<!-- Fast spring -->
    	<HoverTilt springOptions={{ stiffness: 0.5, damping: 0.2 }}>
    	  <div class="card">stiffness: 0.5, damping: 0.6</div>
    	</HoverTilt>

    	<!-- Slow spring -->
    	<HoverTilt springOptions={{ stiffness: 0.05, damping: 0.9 }}>
    	  <div class="card">stiffness: 0.1, damping: 0.9</div>
    	</HoverTilt>
    	```
    </TabItem>
    <TabItem label="Web Component">
    	```html
    	<!-- Default spring -->
    	<hover-tilt>
    	  <div class="card">Default</div>
    	</hover-tilt>

    	<!-- Fast spring (requires JSON string) -->
    	<hover-tilt spring-options='{"stiffness": 0.5, "damping": 0.2}'>
    	  <div class="card">stiffness: 0.5, damping: 0.6</div>
    	</hover-tilt>

    	<!-- Slow spring -->
    	<hover-tilt spring-options='{"stiffness": 0.05, "damping": 0.9}'>
    	  <div class="card">stiffness: 0.1, damping: 0.9</div>
    	</hover-tilt>
    	```
    </TabItem>

</Tabs>

## Exit Delay

Control how long the component waits before animating back to the default
state when the cursor leaves. Use `exitDelay` (in milliseconds) to
create a smoother experience.

<Tabs>
	<TabItem label="Svelte">
		```svelte

    	<!-- Default exit delay (200ms) -->
    	<HoverTilt>
    	  <div class="card">Default</div>
    	</HoverTilt>

    	<!-- Longer exit delay -->
    	<HoverTilt exitDelay={1000}>
    	  <div class="card">exitDelay: 1000ms</div>
    	</HoverTilt>

    	<!-- No exit delay -->
    	<HoverTilt exitDelay={0}>
    	  <div class="card">exitDelay: 0ms</div>
    	</HoverTilt>
    	```
    </TabItem>
    <TabItem label="Web Component">
    	```html
    	<!-- Default exit delay (200ms) -->
    	<hover-tilt>
    	  <div class="card">Default</div>
    	</hover-tilt>

    	<!-- Longer exit delay -->
    	<hover-tilt exit-delay="1000">
    	  <div class="card">exit-delay: 1000ms</div>
    	</hover-tilt>

    	<!-- No exit delay -->
    	<hover-tilt exit-delay="0">
    	  <div class="card">exit-delay: 0ms</div>
    	```

## Blend Modes

Change how the glare effect blends with the content using the `blendMode` prop. Common values include `overlay`, `screen`, `multiply`, `soft-light`, and more.

<Tabs>
	<TabItem label="Svelte">
		```svelte

    	<!-- Default blend mode (overlay) -->
    	<HoverTilt>
    	  <div class="card">Default (overlay)</div>
    	</HoverTilt>

    	<!-- Color dodge blend mode -->
    	<HoverTilt blendMode="color-dodge">
    	  <div class="card">blendMode: color-dodge</div>
    	</HoverTilt>

    	<!-- Plus lighter blend mode -->
    	<HoverTilt blendMode="plus-lighter">
    	  <div class="card">blendMode: plus-lighter</div>
    	</HoverTilt>
    	```

    <TabItem label="Web Component">
